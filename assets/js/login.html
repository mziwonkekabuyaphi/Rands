/** * assets/js/login.js — Rands Vibe * * Handles UI interactions, form validation, and user input * Authentication logic is delegated to authService.js */ import { signIn, getUserRole, getCurrentSession, redirectByRole } from './authService.js'; // DOM references const emailInput = document.getElementById('email'); const passwordInput = document.getElementById('password'); const loginBtn = document.getElementById('loginBtn'); const loginBtnLabel = document.getElementById('loginBtnLabel'); const authError = document.getElementById('authError'); const togglePwBtn = document.getElementById('togglePw'); const eyeIcon = document.getElementById('eyeIcon'); const eyeOffIcon = document.getElementById('eyeOffIcon'); // UI helpers function showFieldError(fieldId, message) { const errEl = document.getElementById(`err-${fieldId}`); const inputEl = document.getElementById(fieldId); if (errEl) { if (message) errEl.textContent = message; errEl.classList.add('visible'); } if (inputEl) inputEl.classList.add('has-error'); } function hideFieldError(fieldId) { const errEl = document.getElementById(`err-${fieldId}`); const inputEl = document.getElementById(fieldId); if (errEl) errEl.classList.remove('visible'); if (inputEl) inputEl.classList.remove('has-error'); } function showAuthError(message) { if (!authError) return; authError.classList.remove('visible'); authError.offsetHeight; // force reflow authError.textContent = message; authError.classList.add('visible'); } function hideAuthError() { if (authError) authError.classList.remove('visible'); } function setLoading(isLoading) { if (!loginBtn || !loginBtnLabel) return; loginBtn.disabled = isLoading; loginBtn.classList.toggle('loading', isLoading); loginBtnLabel.textContent = isLoading ? 'SIGNING IN…' : 'SIGN IN'; } // Password visibility toggle function setupPasswordToggle() { if (!togglePwBtn || !passwordInput || !eyeIcon || !eyeOffIcon) return; togglePwBtn.addEventListener('click', () => { const isPassword = passwordInput.type === 'password'; passwordInput.type = isPassword ? 'text' : 'password'; eyeIcon.style.display = isPassword ? 'none' : ''; eyeOffIcon.style.display = isPassword ? '' : 'none'; togglePwBtn.setAttribute('aria-label', isPassword ? 'Hide password' : 'Show password'); }); } // Form validation const EMAIL_RE = /^[^\s@]+@[^\s@]+\.[^\s@]+$/; function validateForm() { let valid = true; // Email validation if (!emailInput?.value?.trim() || !EMAIL_RE.test(emailInput.value.trim())) { showFieldError('email', 'Enter a valid email address'); valid = false; } else { hideFieldError('email'); } // Password validation if (!passwordInput?.value) { showFieldError('password', 'Password is required'); valid = false; } else { hideFieldError('password'); } return valid; } // Session check on page load async function checkExistingSession() { const { session, error } = await getCurrentSession(); if (error) { console.warn('Session check error:', error); return; } if (session?.user) { const { role } = await getUserRole(session.user.id); redirectByRole(role); } } // Login handler async function handleLogin() { hideAuthError(); if (!validateForm()) return; const email = emailInput.value.trim(); const password = passwordInput.value; setLoading(true); const { user, error } = await signIn(email, password); if (error || !user) { showAuthError(error || 'Login failed. Please try again.'); setLoading(false); return; } // Success — fetch role and redirect const { role } = await getUserRole(user.id); redirectByRole(role); } // Clear errors on input function setupInlineErrorClearing() { if (emailInput) { emailInput.addEventListener('input', () => { hideFieldError('email'); hideAuthError(); }); } if (passwordInput) { passwordInput.addEventListener('input', () => { hideFieldError('password'); hideAuthError(); }); } } // Enter key submits form function setupEnterKey() { const loginForm = document.getElementById('loginForm'); if (!loginForm) return; loginForm.addEventListener('keydown', (e) => { if (e.key === 'Enter' && !loginBtn?.disabled) { e.preventDefault(); handleLogin(); } }); } // Initialize the login page async function init() { await checkExistingSession(); setupPasswordToggle(); setupInlineErrorClearing(); setupEnterKey(); if (loginBtn) { loginBtn.addEventListener('click', handleLogin); } } // Start the application init();
